exports.getPath = function (url, params) {
  if (!url) {
    return false;
  }
 
  for (let key in params) {
    if (!params[key] || params[key] === "") {
      let regex = new RegExp("([?|&])" + key + "=:" + key);
      url = url.replace(regex, "");
    } else {
      url = url.replace(":" + key, params[key]);
    }
  }
  return url;
};
